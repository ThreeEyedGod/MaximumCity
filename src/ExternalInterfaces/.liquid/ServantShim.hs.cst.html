 32:    = APIGatewayProxyRequest (Embedded Value) -> IO (APIGatewayProxyResponse (Embedded Value))

 34: -- | Constructs a APIGatewayHandler for a request to this AWS Lambda function.
 35: --
 36: -- A handler converts the APIGatewayProxyRequest into a regular WAI Request
 37: -- value. This is then passed into the Servant application and handled as if it
 38: -- was a regular Servant request.
 39: --
 40: -- The passed Application is used to handle all requests coming into the handler.
 41: makeHandler :: Application -> APIGatewayHandler
 42: makeHandler application apiGatewayRequest = do
 43:   responseRef <- newIORef Protolude.undefined
 44:   request <- flip proxyToRequest apiGatewayRequest <$> newIORef reqBody
 45:   _ <- application request $ streamResponseTo responseRef
 46:   result <- readIORef responseRef
 47:   pure . responseFromBody . decode $ result
 48:   where
 49:     reqBody =
 50:       maybe mempty (BL.toStrict . encode) $
 51:       apiGatewayRequest ^. requestBodyEmbedded
 52: 
 53: -- | Streams an HTTP response into an IORef, confirming receipt of the response.
 54: --
 55: -- This is intended to be used to construct the second argument to the WAI
 56: -- Application type, a (Response -> IO ResponseReceived) function. This allows
 57: -- us to extract the response from a WAI application without sending that
 58: -- response to the client.
 59: streamResponseTo :: IORef BL.ByteString -> Response -> IO ResponseReceived
 60: streamResponseTo responseRef resp = do
 61:   body <-
 62:     case resp of
 63:       ResponseBuilder _ _ builder -> pure $ toLazyByteString builder
 64:       _ ->
 65:         throwIO $
 66:         IOError
 67:           Nothing
 68:           OtherError
 69:           ""
 70:           "unable to process response"
 71:           Nothing
 72:           Nothing
 73:   writeIORef responseRef body
 74:   pure ResponseReceived
 75: 
 76: addCorsHeaders :: ResponseHeaders -> ResponseHeaders
 77: addCorsHeaders headers = allowOrigin : allowCredentials : headers
 78:   where
 79:     allowOrigin = ("Access-Control-Allow-Origin", "*")
 80:     allowCredentials = ("Access-Control-Allow-Credentials", "true")

 82: responseFromBody :: Maybe Value -> APIGatewayProxyResponse (Embedded Value)
 83: --responseFromBody Nothing = responseBadRequest -- telegram repeatedly sends request if it was sent this
 84: responseFromBody Nothing = responseOK
 85: responseFromBody (Just body) =
 86:   responseOK & responseBodyEmbedded ?~ body & agprsHeaders %~ addCorsHeaders

 88: proxyToRequest ::
 89:      IORef B.ByteString -> APIGatewayProxyRequest (Embedded Value) -> Request
 90: proxyToRequest bodyRef apiGWRequest =
 91:   defaultRequest
 92:     { requestMethod = apiGWRequest ^. agprqHttpMethod
 93:     , httpVersion = http11
 94:     , rawPathInfo = rawPath
 95:     , rawQueryString = query
 96:     , requestHeaders = apiGWRequest ^. agprqHeaders
 97:     , pathInfo = pathParts

 99:     , Network.Wai.Internal.requestBody = ioBody

105:     query = renderQuery True $ apiGWRequest ^. agprqQueryStringParameters
